有一个文件夹，存在多个子文件夹，每个文件夹中还有多个 Excel 表格文件，每个表格文件还有多个 sheet。这些 Excel 表格中，有一部分 Excel 表格是说明有什么活动并说明这个活动是什么类型的，每参加一个活动会获得相关活动的分数，其他表格中说明有什么人参加，请写一段 python 代码帮我查找某一个特定的人参加的活动，打印出这个人参加了哪些活动，并将类型的分数一并答应
```python
import os
import pandas as pd

# 定义要查找的人的名字
person_name = '张三'

# 定义存放Excel文件的文件夹路径
folder_path = 'path_to_your_folder'

# 定义一个空的DataFrame来存储结果
activities_df = pd.DataFrame(columns=['活动名称', '活动类型', '分数'])

# 遍历文件夹中的所有文件
for root, dirs, files in os.walk(folder_path):
    for file in files:
        if file.endswith('.xlsx') or file.endswith('.xls'):
            # 构建完整的文件路径
            file_path = os.path.join(root, file)
            # 使用pandas读取Excel文件
            for sheet_name in pd.ExcelFile(file_path).sheet_names:
                df = pd.read_excel(file_path, sheet_name=sheet_name)
                
                # 假设有一个sheet记录活动和分数，我们将其命名为'Activities'
                if sheet_name == 'Activities':
                    # 存储活动和分数信息
                    activities = df[['活动名称', '活动类型', '分数']]
                
                # 假设另一个sheet记录人员参与的活动，我们将其命名为'Participants'
                elif sheet_name == 'Participants':
                    # 查找特定人员参加活动的记录
                    participant_activities = df[df['参与者'] == person_name]
                    # 如果找到了记录，将活动名称添加到activities_df中
                    if not participant_activities.empty:
                        for index, row in participant_activities.iterrows():
                            activity_name = row['活动名称']
                            # 在活动和分数表中查找对应的活动类型和分数
                            activity_info = activities[activities['活动名称'] == activity_name]
                            if not activity_info.empty:
                                # 将找到的信息添加到结果DataFrame中
                                activities_df = activities_df.append({
                                    '活动名称': activity_name,
                                    '活动类型': activity_info['活动类型'].values[0],
                                    '分数': activity_info['分数'].values[0]
                                }, ignore_index=True)

# 打印这个人参加了哪些活动以及活动的分数
print(f"{person_name} 参加了以下活动：")
print(activities_df)
```
